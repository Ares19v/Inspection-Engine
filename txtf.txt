# 1. Define the Python training script
$train_script = @"
import os
from roboflow import Roboflow
from ultralytics import YOLO

# 1. Download Dataset from Roboflow
rf = Roboflow(api_key="YOUR_ROBOFLOW_API_KEY")
project = rf.workspace("pcb-yylun").project("pcb-master-dataset")
dataset = project.version(2).download("yolov11")

# 2. Initialize YOLO11 with your GPU (RTX 5060)
# We use the 'small' model (yolo11s) for a balance of speed and accuracy
model = YOLO("yolo11s.pt")

# 3. Start Training
print("Starting Training on RTX 5060...")
results = model.train(
    data=os.path.join(dataset.location, "data.yaml"),
    epochs=50,
    imgsz=640,
    plots=True,
    device=0  # This forces it to use your GPU
)

print("Training Complete! Model saved in 'runs/detect/train/weights/best.pt'")
"@

# 2. Write the script to the file
Set-Content -Path "model_training\train.py" -Value $train_script -Encoding UTF8

# 3. Run the training
python model_training\train.py
